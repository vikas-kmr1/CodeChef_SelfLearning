# cook your dish here
for i in range(int(input())):
    n,k = map(int, input().split())
    
    arr = list( map(int, input().split()))
    
    min_moves  = 2**31 - 1
    player = -1
    for j in range(n):
        if arr[j] <= k:
            if k % arr[j] == 0:
                if k//arr[j] < min_moves:
                    min_moves = k//arr[j]
                    player = arr[j]
    if min_moves < (2**31 - 1):
        print(player)
        continue
    print(-1)
